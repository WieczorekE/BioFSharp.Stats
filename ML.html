The generic schema-reader compiler uses F# types to describe a data schema and the schema compiler helps convert untyped data from separated value text files into this data schema.

In the example the Csv-reader reads the famouse iris data set. An IrisItem type represents a row within the data set.

type IrisItem = {
  [<FieldAttribute("Sepal length")>]SepalLength : float
  [<FieldAttribute("Sepal width")>] SepalWidth : float
  [<FieldAttribute("Petal length")>] PetalLength : float
  [<FieldAttribute("Petal width")>] PetalWidth : float
  [<FieldAttribute("Species")>] Species : string }


let toPlotlyString (c:Colors.Color) =
  let a,r,g,b = Colors.toArgb c
  sprintf "rgba(%i,%i,%i,%i)" r g b a 


let path = (__SOURCE_DIRECTORY__) + "\\data\\irisData.csv" //....
let hasHeader = true
let separator = ','
let irisData = 
  Seq.fromFileWithCsvSchema<IrisItem>(path,
    separator,hasHeader,schemaMode = SchemaMode.Fill)
  |> Seq.toList

let x = irisData |> Seq.map (fun x -> x.SepalLength)
let y = irisData |> Seq.map (fun y -> y.PetalLength)
let l =
  irisData
  |> Seq.map
    (fun l -> match l.Species with
              | "Iris-setosa"     -> Colors.Table.Office.red 
              | "Iris-versicolor" -> Colors.Table.Office.green
              | "Iris-virginica"  -> Colors.Table.Office.lightBlue
              | _                 -> Colors.Table.Office.grey
              |> toPlotlyString
    )

open BioFSharp.Stats.ML
open BioFSharp.Stats.ML.Unsupervised

let irisFeatures = 
    irisData 
    |> List.map 
           (fun ii -> 
           [ ii.SepalLength; ii.SepalWidth; ii.PetalLength; ii.PetalWidth ])
let irisLables = irisData |> List.map (fun ii -> ii.Species)
let irisFeaturesMatrix = DenseMatrix.ofColumnList irisFeatures


let adjCenter = PCA.toAdjustCenter irisFeaturesMatrix
let irisPCA = PCA.compute adjCenter irisFeaturesMatrix
let irisDataPCA = PCA.transform adjCenter irisPCA irisFeaturesMatrix
let irisrev = PCA.revert adjCenter irisPCA irisDataPCA

let irisPcaLabel =
  [ "Sepal length"; "Sepal width"; "Petal length"; "Petal width" ]
